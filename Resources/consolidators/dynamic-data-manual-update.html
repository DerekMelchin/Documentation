#  It's only currently possible to make a history request for custom/alt data in Python that returns a DataFrame
#  To update the consolidator, iterate through the DataFrame, then use the data of each row to create the data objects and update the consolidator.
#  See https://github.com/QuantConnect/Lean/issues/6927
// Example 1: Update the consolidator with data from the Slice object
public override void OnData(Slice slice)
{
    if slice.ContainsKey(_symbol)
    {
        _consolidator.Update(slice[_symbol]);
    }
}

{
    _consolidator.Update(data);
